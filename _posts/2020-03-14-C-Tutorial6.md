## 1 结构体变量定义、引用、初始化

要定义一个结构体变量，我们把结构体名看做一个类型名就可以了。

如何用结构体类型

```c
struct date
{
  int year;
  int month;
  int day;
}

struct student
{
  int num;
  int age;
  char name[100];
  date birthday;
}；

int main()
{
  struct student s1,s2;  //1. struct 结构体名  变量名列表
  //2. 定义结构类型的时候，顺便定义了变量
  //struct 结构体名
  //{
  //  成员列表
  //}变量名列表
  //3. 直接定义结构体变量，连结构体名也省略（不建议），后续只能用那几个结构体变量
    
}
```

结构体类型变量的引用

```c
s1.num = 1001  // . 叫结构成员运算符，它优先级和括号平级的，所以可以把s1.num看成一个整体
s1.age++;  //因为.优先级最高，年龄自加
int *p = &s1.num;
printf("%d\n",*p)
```

结构体变量的初始化

```c
struct student s3 = [1101,20,"王五",2000,10,20];
```

## 2 结构体数组、结构体指针

### 2.1 结构体数组

```c
// 单独写一行定义
struct student stu[3];

struct student
{
  int num;

  char name[100];

struct student stu[3] =
```

### 2.2 结构体指针

```c
struct student stu;
struct student *p;
p  = &stu;
stu.age  =16;
// 通过结构体指针访问结构体变量
(*p).age = 29;
p->age = 20;

struct student stus[3] = 
  [[1101,20,"王五",2000,10,20],
  [1102,20,"张三",2000,10,20],
  [1103,20,"李四",2000,10,20]];
struct student *ps = stus;
// 指向结构体数组的指针
for(int i = 0; i < 3; +++i)
{
  printf("num=%d\n",ps->num)
  ps++;  // ps 指向了下一个数组元素的开始地址
}
int ilen = sizeof(struct student);

// 说明
printf("num=%d\n",(++ps)->num);
printf("num=%d\n",++ps->num);

// 不能让ps指向某一数组元素中的成员 ???
//ps = &stu[0].num; 右边是整型地址
ps = stu[0]; // 可以
```

#### 2.3 用指向结构体的指针做函数参数

### 3. 共用体、枚举类型、typedef

#### 3.1 共用体 （联合）

把几种不同类型的变量存放到同一段内存单元。

几个变量占同一段内存的结构。

```c
#include <string.h>
//union 共用体名
//{
//	成员列表
//}变量列表;

union myuni
{
  int carnum;
  char cartype;
  char cname[60];
//}a,b,c; //将共用体定义和共用体变量定义放一起
}

int main()
{
  union myuni a,b,c;
  // 结构体和共用体明显不同
  int ilen(struct student);  // 结构体变量占的内存是各个变量站的内存之和
  													 // 有时候会多分配几个字节，保证字节对齐，为了考虑运行效率
  int ilen(union myuni);	// 共用体所占的内存长度是最长的成员长度
  
  // 说明
  // a)
  a.cname;
  a.carnum;
  // b) 共用体变量的特点
  // 每一个瞬间只能存放其中一种，每个瞬间只能有一个成员起作用，其他成员不起作用
  // 程序代码中最后个那个成员赋值，那个成员就起作用
  a.carnum = 321434354;
  strcpy(a.cname,"adf");
  printf("%s\n", a.cname);
  // c) 共用体变量的地址和其成员的地址都相同；
  // &a &a.cname &a.cartype
  // d) 共用体变量不能在定义的时候初始化
  union myuni aa = [12]; // okay
  union myuni aa = ['d']; // fail
  // e) 不能把共用体变量作为函数参数，也不能让函数带回共用体变量 ？？
}
```

#### 3.2 枚举类型

```c
// 四种颜色：red, green, blue, yellow
enum color
{
  Red, // 枚举常量（可以当成一个整型数字）= 0， 看成整型常量，占4个字节
  Green,
  Blue,
  Yellow
};

int main()
{
  // mycolor1, mycolor2的值限于列举的值的范围内
  enum color mycolor1, mycolor2;
  // 可以直接定义枚举变量，不用枚举名
  enum
	{
 	  Red, // 枚举常量（可以当成一个整型数字）
 	 	Green,
	  Blue,
	  Yellow
	}mycolor1, mycolor2;
  // red, green, blue, yellow 是 枚举“常量”
  Red = 30; // fail
  // 可以直接给枚举变量赋值
  mycolor1 = Yellow;
  // 可以在定义时，定义枚举常量的值
  enum color
	{
  	Red = -8;
	  Green,
	  Blue = 30,
	  Yellow
	};
  // 枚举值就是整型值
  mycolor1 = (enum color) -8;
  mycolor1 = (enum color) 2000;
  printf("%d\n",mycolor1); // 2000
  // 枚举值可以赋给一个整型
  int a = mycolor1;
}

```

#### 3.3 typedef

定义新的类型名，来代替已经有的类型名

```c
typedef int INTEGER;

typedef struct date
{
  ...
}DATE;

DATE birthday;  // 不需要写struct date了

// 变形
typedef int NUM[100];  // 定义NUM为整型数组类型
NUM n; // 等价于 int n[100]

typedef char *PSTRING; // 定义PSTRING为字符指针类型
PSTRING p; // char *p, *q;
typedef int (*POINTER)() // 指向函数的指针类型，该函数返回整型值
```

总结，以定义一个整型数组为例

```c
// step1：写出常规的定义整型数组的方法
int n[100];
// step2：将变量名n替换成自己想用的类型名
int NUM[100];
// step3：在前面加上typedef
typedef int NUM[100];
```

说明：

- 类型别名一般大写

- 用来定义类型名的，不是定义变量的

- 没有创造新类型

- typedef是编译时候处理的

  生成可执行文件：

  预处理【#define,#include,#ifdef】-->编译【typedef】--> 汇编 --> 链接

- 主要作用：程序的通用性和可移植性
